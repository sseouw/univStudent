EX3-1


PROCESSOR 16F876A
#INCLUDE <P16F876A.INC>

VARIABLE	LED1 = 20H
VARIABLE	DBUF1 = 22H
VARIABLE	DBUF2 = 23H

ORG	0000
BSF	STATUS, RP0
CLRF	TRISA
CLRF	TRISC
MOVLW	B'00000111'
MOVWF	ADCON1
BCF	STATUS, RP0

MOVLW	01
ADDLW	00
MOVWF	LED1

LOOP
	MOVF	LED1, W
	MOVWF	PORTC
	RLF	LED1, F
	CALL 	DELAY
	GOTO	LOOP
	
DELAY
	MOVLW	.125
	MOVWF	DBUF1
LP1	MOVLW	.200
	MOVWF	DBUF2
LP2	NOP
	DECFSZ	DBUF2, F
	GOTO	LP2
	DECFSZ	DBUF1, F
	GOTO	LP1
	RETURN
	
	END

PROCESSOR 16F876A
#INCLUDE <P16F876A.INC>

VARIABLE	LED1 = 20H
VARIABLE	DBUF1 = 22H
VARIABLE	DBUF2 = 23H

ORG	0000
BSF	STATUS, RP0
MOVLW	B'00000001'		//이부분 주의!
MOVWF	TRISA
CLRF	TRISC
MOVLW	B'00000111'
MOVWF	ADCON1
BCF	STATUS, RP0

MOVLW	01
ADDLW	00
MOVWF	LED1

LOOP
	MOVF	LED1, W
	XORLW	0FFH		//출력 반전
	MOVWF	PORTC
	BTFSC	PORTA,0
	GOTO	LP3
	GOTO	LP4
	
LP3	RLF	LED1, F
	CALL 	DELAY
	GOTO	LOOP
LP4	RRF	LED1, F
	CALL 	DELAY
	GOTO 	LOOP
	
DELAY
	MOVLW	.125
	MOVWF	DBUF1
LP1	MOVLW	.200
	MOVWF	DBUF2
LP2	NOP
	DECFSZ	DBUF2, F
	GOTO	LP2
	DECFSZ	DBUF1, F
	GOTO	LP1
	RETURN
	
	END

PROCESSOR 16F876A
#INCLUDE <P16F876A.INC>

VARIABLE	LED1 = 20H
VARIABLE	LED2 = 21H
VARIABLE	DBUF1 = 22H
VARIABLE	DBUF2 = 23H

ORG	0000
BSF	STATUS, RP0
CLRF	TRISA
CLRF	TRISC
MOVLW	B'00000111'
MOVWF	ADCON1
BCF	STATUS, RP0

LABEL	CLRF	LED2
	MOVLW	01
	ADDLW	00
	MOVWF	LED1
	

LOOP
	BTFSC	LED2, 4
	GOTO	LABEL

	MOVF	LED1, W
	MOVWF	PORTC
	MOVF	LED2, W
	MOVWF	PORTA
	RLF	LED1, F
	RLF	LED2, F

	CALL 	DELAY
	GOTO	LOOP
	
DELAY
	MOVLW	.125
	MOVWF	DBUF1
LP1	MOVLW	.200
	MOVWF	DBUF2
LP2	NOP
	DECFSZ	DBUF2, F
	GOTO	LP2
	DECFSZ	DBUF1, F
	GOTO	LP1
	RETURN
	
	END
